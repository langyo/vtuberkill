    Example 
    let _x = 0;
let onUpdateX = new Subject();
set x(x){
    console.log('x: '+_x+'to '+ x);
    _x = x;
    onUpdateX.notify(x);
}
get x(){
    return _x;
}
let y = 0;
onUpdateX.set(undefined, function(x){
    console.log('y: '+y+'to '+x);
    y = x;
});
x = 1;
console.log('x: '+x+'y: '+y);
x = 'Hello Subject';
console.log('x: '+x+'y: '+y);
